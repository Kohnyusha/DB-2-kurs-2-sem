USE master;
CREATE database k_MyBase2;

use k_MyBase2

CREATE table Студенты 
(
	Номер_зачетки int primary key,
	Фамилия nchar(30) NOT NULL,
	Имя nchar(30) NOT NULL,
	Отчество nchar(30) NOT NULL,
	Адрес nchar(30) NOT NULL,
	Телефон nchar(20) NOT NULL,
)
CREATE table Предметы 
(
	Название_предмета nchar(50) primary key,
	Объем_лекций int NOT NULL,
	Объем_практических int,
	Объем_лабораторных int,
)
CREATE table Оценки 
(
	Номер_записи_в_таблице int primary key,
	Название_предмета nchar(50) foreign key references Предметы(Название_предмета),
	Номер_билета_ int NOT NULL,
	Отметка int NOT NULL,
	Номер_зачетки int foreign key references Студенты(Номер_зачетки),
	
)
